<script type="text/x-dc" data-dc-script>
class Component extends DCLogic {
  state = { activeCat: 'all', sort: 'destaques' };

  cats = [
    { key: 'all', label: 'Todos' },
    { key: 'airport', label: 'Aeroporto', color: '--c-airport' },
    { key: 'seguro', label: 'Seguros', color: '--c-seguro' },
    { key: 'viagem', label: 'Viagem', color: '--c-viagem' },
    { key: 'cashback', label: 'Cashback', color: '--c-cashback' },
    { key: 'compras', label: 'Compras', color: '--c-compras' },
  ];

  sorts = [
    { key: 'destaques', label: 'Destaques' },
    { key: 'categoria', label: 'Categoria' },
  ];

  all = [
    { category: 'airport', via: 'Nubank Ultravioleta', title: '4 acessos Priority Pass por ano', desc: '+1.700 salas VIP em 145 países.', isNew: true, originType: 'emissor', originLabel: 'Emissor · Nubank' },
    { category: 'seguro', via: 'XP Infinite', title: 'Seguro p/ veículos de locadora', desc: 'Cobertura ao pagar a locação com o cartão.', tag: 'Seguros', originType: 'bandeira', originLabel: 'Bandeira · Visa Infinite' },
    { category: 'viagem', via: 'C6 Carbon', title: 'Seguro-viagem internacional', desc: 'Até US$ 100 mil em despesas médicas.', tag: 'Viagem', originType: 'emissor', originLabel: 'Emissor · C6' },
    { category: 'compras', via: 'Inter Prime', title: 'Garantia Estendida Mastercard Black', desc: '+1 ano sobre a garantia do fabricante.', tag: 'Compras', originType: 'parceiro', originLabel: 'Parceiro' },
    { category: 'cashback', via: 'Itaú Personnalité', title: '5% de cashback em viagens', desc: 'Crédito na fatura do mês seguinte.', tag: 'Cashback', originType: 'emissor', originLabel: 'Emissor · Itaú' },
    { category: 'airport', via: 'Bradesco Aeternum', title: 'GRU Lounge — Terminal 3', desc: '2 acessos por mês para você e convidado.', tag: 'Aeroporto', originType: 'parceiro', originLabel: 'Parceiro · GRU' },
  ];

  renderVals() {
    const { activeCat, sort } = this.state;

    const chips = this.cats.map((c) => ({
      label: c.label,
      dotCls: c.color ? `cat-${c.key}` : null,
      cls: c.key === activeCat ? 'chip on' : 'chip',
      pressed: c.key === activeCat,
      select: () => this.setState({ activeCat: c.key }),
    }));

    const segs = this.sorts.map((s) => ({
      label: s.label,
      cls: s.key === sort ? 'on' : '',
      selected: s.key === sort,
      select: () => this.setState({ sort: s.key }),
    }));

    let passes = activeCat === 'all' ? this.all.slice() : this.all.filter((p) => p.category === activeCat);
    if (sort === 'categoria') passes = passes.sort((a, b) => a.category.localeCompare(b.category));

    return {
      chips,
      segs,
      passes,
      isEmpty: passes.length === 0,
      toggleTheme: () => {
        const el = document.documentElement;
        const next = el.getAttribute('data-theme') === 'dark' ? 'light' : 'dark';
        el.setAttribute('data-theme', next);
        try { localStorage.setItem('mb-theme', next); } catch (e) { /* noop */ }
      },
    };
  }
}

/* ---- ds-base.js (carrega styles.css + bundle, define tema inicial) ---- */
// Loads this design system into the template. In a consuming project, point
// base at the bound DS folder relative to this file.
(() => {
  const base = '../..';
  for (const p of ["styles.css"]) {
    const l = document.createElement('link');
    l.rel = 'stylesheet'; l.href = base + '/' + p;
    document.head.appendChild(l);
  }
  // tema: preferência salva > preferência do sistema
  try {
    const saved = localStorage.getItem('mb-theme');
    const sys = window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light';
    document.documentElement.setAttribute('data-theme', saved || sys);
  } catch (e) { /* noop */ }

  const s = document.createElement('script');
  s.src = base + '/_ds_bundle.js';
  document.head.appendChild(s);
})();
